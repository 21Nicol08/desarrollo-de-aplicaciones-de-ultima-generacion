using System;

class epa
{
    static void Main()
    {
        Console.Write("Ingresa tu nombre: ");
        string nombre = Console.ReadLine();
        Console.WriteLine("Hola, " + nombre + "!");
    }
}

using System;

class epa
{
    static void Main()
    {
        Console.Write("Ingresa tu nombre: ");
        string nombre = Console.ReadLine();
        Console.ForegroundColor = ConsoleColor.Red;
        Console.WriteLine("Hola, " + nombre + " ❤️");
        Console.ResetColor();
    }
}
